### 2.3 app/core/story_engine.py

剧情引擎核心

```python
from app.services.openai_service import OpenAIService
from app.services.thread_service import ThreadService
from app.services.memory_service import MemoryService
from app.core.agents.director import DirectorAgent
from app.core.agents.writer import WriterAgent
from app.core.validator import Validator

class StoryEngine:
    def __init__(self):
        self.openai_service = OpenAIService()
        self.thread_service = ThreadService()
        self.memory_service = MemoryService()
        self.director = DirectorAgent()
        self.writer = WriterAgent()
        self.validator = Validator()
    
    async def process_input(self, thread_id: str, user_input: str):
        # 1. 获取 Thread
        thread = await self.thread_service.get_thread(thread_id)
        
        # 2. 获取历史上下文（RAG）
        context = await self.memory_service.get_relevant_context(
            thread_id, user_input
        )
        
        # 3. Director 规划剧情
        story_plan = await self.director.plan(thread_id, user_input, context)
        
        # 4. Writer 生成内容
        story_content = await self.writer.generate(
            thread_id, story_plan, context
        )
        
        # 5. 验证
        validated_content = await self.validator.validate(
            story_content, thread_id
        )
        
        # 6. 更新记忆
        await self.memory_service.update_memory(
            thread_id, validated_content
        )
        
        return validated_content
```

### 2.4 app/services/openai_service.py

OpenAI API 封装

```python
from openai import OpenAI
from app.core.config import settings
import logging

logger = logging.getLogger(__name__)

class OpenAIService:
    def __init__(self):
        self.client = OpenAI(api_key=settings.openai_api_key)
        self.model = settings.openai_model
    
    async def create_assistant(self, name: str, instructions: str, tools: list):
        """创建 Assistant"""
        assistant = self.client.beta.assistants.create(
            name=name,
            instructions=instructions,
            model=self.model,
            tools=tools
        )
        return assistant
    
    async def create_thread(self):
        """创建 Thread"""
        thread = self.client.beta.threads.create()
        return thread
    
    async def add_message(self, thread_id: str, content: str, role: str = "user"):
        """添加消息到 Thread"""
        message = self.client.beta.threads.messages.create(
            thread_id=thread_id,
            role=role,
            content=content
        )
        return message
    
    async def run_assistant(self, thread_id: str, assistant_id: str):
        """运行 Assistant"""
        run = self.client.beta.threads.runs.create(
            thread_id=thread_id,
            assistant_id=assistant_id
        )
        
        # 等待完成
        while run.status in ["queued", "in_progress"]:
            run = self.client.beta.threads.runs.retrieve(
                thread_id=thread_id,
                run_id=run.id
            )
        
        if run.status == "completed":
            messages = self.client.beta.threads.messages.list(
                thread_id=thread_id
            )
            return messages.data[0]
        else:
            raise Exception(f"Run failed with status: {run.status}")
    
    async def generate_image(self, prompt: str, size: str = "1024x1024"):
        """生成图像"""
        response = self.client.images.generate(
            model=settings.openai_image_model,
            prompt=prompt,
            size=size,
            quality="standard",
            n=1
        )
        return response.data[0].url
    
    async def create_embedding(self, text: str):
        """创建向量嵌入"""
        response = self.client.embeddings.create(
            model=settings.openai_embedding_model,
            input=text
        )
        return response.data[0].embedding
```

### 2.5 app/core/agents/director.py

Director Agent

```python
from app.core.agents.base import BaseAgent
from app.services.openai_service import OpenAIService
from app.core.config import settings

class DirectorAgent(BaseAgent):
    def __init__(self):
        super().__init__()
        self.openai_service = OpenAIService()
        self.assistant_id = settings.director_assistant_id
    
    async def plan(self, thread_id: str, user_input: str, context: str):
        """规划剧情"""
        # 构建 Prompt
        prompt = f"""
        当前剧情上下文：
        {context}
        
        玩家输入：
        {user_input}
        
        请分析并规划下一步剧情。
        """
        
        # 运行 Assistant
        result = await self.openai_service.run_assistant(
            thread_id, self.assistant_id
        )
        
        # 解析结果（JSON 格式）
        import json
        plan = json.loads(result.content[0].text.value)
        
        return plan
```

## 三、数据库模型

### 3.1 app/models/database.py

```python
from sqlalchemy import Column, String, JSON, DateTime, ForeignKey
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import relationship
import uuid
from datetime import datetime

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    username = Column(String(255))
    created_at = Column(DateTime, default=datetime.utcnow)
    
    threads = relationship("Thread", back_populates="user")

class Thread(Base):
    __tablename__ = "threads"
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    user_id = Column(UUID(as_uuid=True), ForeignKey("users.id"))
    openai_thread_id = Column(String(255), unique=True)
    created_at = Column(DateTime, default=datetime.utcnow)
    
    user = relationship("User", back_populates="threads")
    story_state = relationship("StoryState", back_populates="thread", uselist=False)
    conversations = relationship("Conversation", back_populates="thread")

class StoryState(Base):
    __tablename__ = "story_states"
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    thread_id = Column(UUID(as_uuid=True), ForeignKey("threads.id"), unique=True)
    current_scene = Column(String(255))
    story_flags = Column(JSON)
    character_relations = Column(JSON)
    emotion_values = Column(JSON)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    thread = relationship("Thread", back_populates="story_state")

class Conversation(Base):
    __tablename__ = "conversations"
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    thread_id = Column(UUID(as_uuid=True), ForeignKey("threads.id"))
    role = Column(String(50))  # user | assistant
    content = Column(JSON)
    metadata = Column(JSON)
    created_at = Column(DateTime, default=datetime.utcnow)
    
    thread = relationship("Thread", back_populates="conversations")
```
