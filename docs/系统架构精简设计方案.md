# 系统架构精简设计方案

> 4 组件架构：ScriptEngine + WorldSimulator + ConsistencyAgent + Orchestrator
> 更新日期: 2026-06-18

---

## 一、设计原则

```
1. ScriptEngine 是唯一的大脑
   → 生成所有交互数据（场景/台词/选项/数值变化）
   → 管理叙事节奏和事件生命周期
   → 所有需要 LLM 的任务都在这里

2. 其他组件都是执行器
   → 只执行 ScriptEngine 的指令
   → 不做决策，只做计算和校验

3. 去掉冗余 Agent
   → DirectorAgent: 职责已移入 ScriptEngine
   → EventAgent: 职责已移入 ScriptEngine
   → DialogueAgent: 职责已移入 ScriptEngine
   → EmotionAgent: 简化为 WorldSimulator 的一部分

4. 保留必要的独立组件
   → WorldSimulator: 时间/天气/情绪应用（纯规则）
   → ConsistencyAgent: 输出质量校验（规则 + 可选 LLM）
```

```python
class NarrativePlanner:
    """叙事节拍管理"""

    BEATS = [
        {"name": "开场相识", "phase": "opening", "start": 0, "end": 5,
         "emotion_target": "neutral_warm", "event_pool": ["first_meeting", "casual_chat"]},
        {"name": "关系建立", "phase": "rising", "start": 5, "end": 13,
         "emotion_target": "growing_close", "event_pool": ["shared_activity", "small_conflict"]},
        {"name": "冲突爆发", "phase": "climax", "start": 13, "end": 23,
         "emotion_target": "intense", "event_pool": ["major_conflict", "critical_choice"]},
        {"name": "冲突化解", "phase": "resolution", "start": 23, "end": 30,
         "emotion_target": "resolution", "event_pool": ["reconciliation", "final_choice"]},
    ]

    def get_current_beat(self, elapsed_minutes: float) -> dict:
        """根据已过时间返回当前节拍"""
        for beat in self.BEATS:
            if beat["start"] <= elapsed_minutes < beat["end"]:
                return beat
        return self.BEATS[-1]

    def get_phase(self, elapsed_minutes: float) -> str:
        """获取当前阶段"""
        return self.get_current_beat(elapsed_minutes)["phase"]
```

#### EndingJudge（结局判断）

```python
class EndingJudge:
    """结局判断"""

    def should_end(self, state: AgentState) -> tuple[bool, str]:
        """判断是否应该结束

        Returns:
            (是否结束, 原因)
        """
        # 硬上限
        if state.world.elapsed_minutes >= 30.0:
            return True, "time_exceeded"
        if state.total_rounds >= 25:
            return True, "rounds_exceeded"

        # 情绪极化（至少 8 轮后）
        if state.total_rounds >= 8:
            emotion = state.emotion

            # 美好结局提前
            if emotion.favorability >= 85 and emotion.trust >= 75:
                return True, "happy_ending_early"

            # 悲剧结局提前
            if emotion.hostility >= 70:
                return True, "hostility_extreme"
            if emotion.favorability <= 15 and emotion.trust <= 20:
                return True, "relationship_collapse"

        return False, "continue"

    def classify_ending(self, emotion: EmotionState) -> str:
        """分类结局类型"""
        if emotion.favorability >= 70 and emotion.hostility <= 20:
            return "happy_ending"
        if emotion.hostility >= 50 or emotion.favorability <= 20:
            return "bad_ending"
        if emotion.favorability >= 40 and emotion.trust >= 50:
            return "neutral_ending"
        return "open_ending"
```

```python
class EventManager:
    """事件生命周期管理"""

    def __init__(self):
        self.event_templates = {
            "first_meeting": "角色与玩家初次相遇",
            "casual_chat": "角色与玩家闲聊日常",
            "shared_activity": "角色与玩家共同完成某件事",
            "small_conflict": "角色与玩家产生小误会",
            "personal_revelation": "角色向玩家透露个人秘密",
            "help_moment": "角色帮助玩家解决一个困难",
            "major_conflict": "角色与玩家发生严重分歧",
            "critical_choice": "角色面临关键抉择",
            "emotional_confession": "角色情感爆发表白",
            "betrayal": "角色感到被背叛",
            "reconciliation": "角色与玩家和解",
            "final_choice": "角色做出最终决定",
            "departure": "角色准备离开",
            "promise": "角色做出承诺",
        }

    def select_event(self, beat: dict, used_events: set, emotion: EmotionState) -> str:
        """选择事件（情绪驱动 + 去重）"""
        import random

        available = [e for e in beat["event_pool"] if e not in used_events]
        if not available:
            available = beat["event_pool"]

        # 情绪驱动偏向
        if emotion.favorability >= 70:
            positive = ["shared_activity", "personal_revelation", "help_moment", "promise"]
            weighted = [e for e in available if e in positive]
            if weighted:
                return random.choice(weighted)

        if emotion.hostility >= 40:
            negative = ["small_conflict", "major_conflict", "betrayal"]
            weighted = [e for e in available if e in negative]
            if weighted:
                return random.choice(weighted)

        return random.choice(available)

    def should_advance_event(self, round_count: int, min_rounds: int = 2,
                             max_rounds: int = 5) -> bool:
        """判断事件是否应该推进"""
        import random

        if round_count < min_rounds:
            return False
        if round_count >= max_rounds:
            return True

        # 渐进式结束
        if round_count >= 3:
            prob = (round_count - 2) * 0.3
            return random.random() < prob

        return False

    def get_event_description(self, event_type: str) -> str:
        """获取事件描述"""
        return self.event_templates.get(event_type, "角色与玩家互动")
```

#### SceneGenerator（场景生成）

```python
class SceneGenerator:
    """场景生成"""

    def __init__(self, scenes_data: dict = None):
        self.scenes_data = scenes_data or {}

    def select_scene(self, current_scene: str, event_type: str,
                     elapsed_minutes: float) -> str:
        """选择场景

        根据事件类型和已过时间决定是否切换场景。
        """
        import random

        # 70% 概率保持当前场景
        if random.random() < 0.7:
            return current_scene

        # 30% 概率切换到新场景
        available = [s for s in self.scenes_data if s != current_scene]
        if available:
            return random.choice(available)

        return current_scene
```

#### DialogueGenerator（台词生成）

```python
class DialogueGenerator:
    """台词生成（LLM）"""

    def __init__(self, llm_tool: LLMTool):
        self.llm = llm_tool

    async def generate(self, state: AgentState, event_description: str) -> str:
        """生成角色台词"""
        prompt = self._build_prompt(state, event_description)

        result = await self.llm.generate(
            prompt, max_tokens=100, temperature=0.9, call_type="dialogue"
        )

        return result if result else "..."

    def _build_prompt(self, state: AgentState, event_description: str) -> str:
        """构建 prompt"""
        emotion = state.emotion
        world = state.world
        personality = state.character_personality.get("keywords", [])

        parts = [
            f"你是{state.character_name}，性格：{'、'.join(personality)}。",
            f"场景：{world.current_scene}，时间：{world.current_time}。",
            f"当前事件：{event_description}。",
            "",
            "你的情绪状态：",
            f"- 好感度：{emotion.favorability:.0f}/100",
            f"- 信任度：{emotion.trust:.0f}/100",
            f"- 快乐：{emotion.happiness:.0f}/100",
            f"- 压力：{emotion.stress:.0f}/100",
        ]

        # 注入长期记忆
        if state.event_summaries:
            parts.append("")
            parts.append("之前的事件：")
            for i, s in enumerate(state.event_summaries[-3:], 1):
                parts.append(f"{i}. {s}")

        if state.knowledge:
            parts.append("")
            parts.append("你对玩家的了解：")
            for k in state.knowledge[:3]:
                parts.append(f"- {k.get('content', '')}")

        # 注入短期记忆
        recent = state.dialogue_history[-6:]
        if recent:
            parts.append("")
            parts.append("最近对话：")
            for msg in recent:
                role = "玩家" if msg.get("role") == "player" else state.character_name
                parts.append(f"{role}: {msg.get('content', '')}")

        parts.append("")
        parts.append(f"请以{state.character_name}的口吻回复一句话（20-80字），符合当前情绪和剧情。")

        return "\n".join(parts)
```

```python
class OptionGenerator:
    """选项生成（LLM + 规则）"""

    def __init__(self, llm_tool: LLMTool):
        self.llm = llm_tool

    async def generate(self, state: AgentState, dialogue: str) -> list:
        """生成玩家选项"""
        # 尝试 LLM 生成
        if self.llm:
            options = await self._generate_with_llm(state, dialogue)
            if options:
                return options

        # 降级：规则生成
        return self._generate_with_rules(state)

    async def _generate_with_llm(self, state: AgentState, dialogue: str) -> list:
        """LLM 生成选项"""
        try:
            prompt = f"""根据以下对话，为玩家生成 3 个选项。

角色台词：{dialogue}
当前情绪：好感度{state.emotion.favorability:.0f}，信任度{state.emotion.trust:.0f}

要求：
1. 每个选项 10-20 字
2. 第一个偏积极，第二个偏中性，第三个偏消极
3. 直接输出 3 行，不要编号"""

            result = await self.llm.generate(
                prompt, max_tokens=100, temperature=0.8, call_type="options"
            )

            if result:
                lines = [l.strip() for l in result.strip().split("\n") if l.strip()]
                if len(lines) >= 3:
                    return [
                        {"id": 0, "text": lines[0], "direction": "positive",
                         "state_changes": {"favorability": 5, "trust": 3, "happiness": 3}},
                        {"id": 1, "text": lines[1], "direction": "neutral",
                         "state_changes": {"favorability": 1, "trust": 1}},
                        {"id": 2, "text": lines[2], "direction": "negative",

        except Exception as e:

             "state_changes": {"favorability": -3, "trust": -2, "stress": 2}},
```

#### EmotionCalculator（数值变化计算）

```python
class EmotionCalculator:
    """数值变化计算"""

    # 叙事阶段修正系数
    STAGE_MULTIPLIER = {
        "opening": 0.6,
        "rising": 1.0,
        "climax": 1.5,
        "resolution": 0.8,
    }

    def calculate_changes(self, option: dict, phase: str,
                          character_personality: dict) -> dict:
        """计算数值变化

        考虑：
        1. 选项的 base state_changes
        2. 叙事阶段修正
        3. 角色性格修正
        """
        base_changes = option.get("state_changes", {})
        multiplier = self.STAGE_MULTIPLIER.get(phase, 1.0)

        # 应用阶段修正
        changes = {k: v * multiplier for k, v in base_changes.items()}

        # 应用性格修正
        changes = self._apply_personality(changes, character_personality)

        return changes

    def _apply_personality(self, changes: dict, personality: dict) -> dict:
        """性格修正"""
        keywords = personality.get("keywords", [])
        if not keywords:
            return changes

        modifiers = {
            "热情": {"favorability": 1.2, "happiness": 1.2},
            "高冷": {"favorability": 0.8, "happiness": 0.8},
            "温柔": {"favorability": 1.1, "trust": 1.1},
        }

        for keyword in keywords:
            if keyword in modifiers:
                for key, mult in modifiers[keyword].items():
                    if key in changes:
                        changes[key] = changes[key] * mult

        return changes
```

### 3.4 核心方法：generate_round

```python
class ScriptEngine:
    async def generate_round(self, state: AgentState) -> RoundOutput:
        """生成一个完整轮次的所有交互数据

        这是 ScriptEngine 的核心方法，一次调用生成所有内容。
        """

        # 1. 叙事节拍
        beat = self.narrative_planner.get_current_beat(state.world.elapsed_minutes)
        phase = beat["phase"]

        # 2. 结局判断
        should_end, end_reason = self.ending_judge.should_end(state)
        if should_end:
            return await self._generate_ending(state, end_reason)

        # 3. 事件管理
        if state.event_round_count == 0 or self.event_manager.should_advance_event(state.event_round_count):
            # 创建新事件
            event_type = self.event_manager.select_event(beat, state.used_event_ids, state.emotion)
            state.pending_event = {
                "type": event_type,
                "description": self.event_manager.get_event_description(event_type),
            }
            state.used_event_ids.add(event_type)
            state.event_round_count = 0

        # 4. 场景生成
        scene = self.scene_generator.select_scene(
            state.world.current_scene,
            state.pending_event.get("type", ""),
            state.world.elapsed_minutes
        )

        # 5. 台词生成（LLM）
        dialogue = await self.dialogue_generator.generate(
            state, state.pending_event.get("description", "")
        )

        # 6. 选项生成（LLM + 规则）
        options = await self.option_generator.generate(state, dialogue)

        # 7. 数值变化计算
        # 注意：这里只计算"默认"变化，实际变化在玩家选择后计算
        # 选项的 state_changes 已经在选项中定义

        # 8. 事件推进判定
        event_ended = self.event_manager.should_advance_event(state.event_round_count)

        return RoundOutput(
            scene=scene,
            character_dialogue=dialogue,
            player_options=options,
            emotion_changes={},  # 实际变化在玩家选择后计算
            event_advance=event_ended,
            event_ended=event_ended,
            phase=phase,
        )

    async def _generate_ending(self, state: AgentState, reason: str) -> RoundOutput:
        """生成结局"""
        ending_type = self.ending_judge.classify_ending(state.emotion)
        name = state.character_name or "角色"

        # 结局台词
        if ending_type == "happy_ending":
            dialogue = f"{name}: 谢谢你...我永远不会忘记今天的。"
        elif ending_type == "bad_ending":
            dialogue = f"{name}: ...（转身离去，没有回头）"
        else:
            dialogue = f"{name}: 今天就这样吧。再见。"

        return RoundOutput(
            scene=state.world.current_scene,
            character_dialogue=dialogue,
            player_options=[],  # 结局无选项
            emotion_changes={},
            event_advance=False,
            event_ended=False,
            phase="ending",
            is_game_finished=True,
            ending_type=ending_type,
        )
```

## 四、WorldSimulator 详细设计

### 4.1 职责

```
WorldSimulator = 纯规则执行器

职责：
  → 时间推进（按阶段步长）
  → 天气变化（循环队列）
  → 情绪变化应用（加法）
  → PAD 向量计算

不负责：
  → 场景切换（已移入 ScriptEngine）
  → 事件选择（已移入 ScriptEngine）
  → 任何 LLM 调用
```

### 4.2 实现

```python
class WorldSimulator:
    """世界模拟器 — 纯规则执行器"""

    PHASE_TIME_INCREMENT = {
        "opening": 1.5, "rising": 2.0, "climax": 2.5, "resolution": 2.0,
    }
    WEATHER_CYCLE = ["clear", "clear", "clear", "cloudy", "cloudy", "rain", "clear"]
    DECAY_RATES = {
        "stress": 0.15, "anxiety": 0.10, "emotion": 0.05,
        "happiness": 0.03, "sadness": 0.08,
    }

    def __init__(self):
        self._weather_index = 0

    def advance_time(self, world: WorldState, phase: str) -> None:
        """推进时间"""
        import random
        increment = self.PHASE_TIME_INCREMENT.get(phase, 2.0)
        world.elapsed_minutes += increment + random.uniform(-0.3, 0.3)
        world.time_of_day_progress = world.elapsed_minutes / 30.0

        if world.time_of_day_progress < 0.3:
            world.current_time = "morning"
        elif world.time_of_day_progress < 0.7:
            world.current_time = "afternoon"
        else:
            world.current_time = "evening"

    def update_weather(self, round_num: int, world: WorldState) -> None:
        """更新天气"""
        if round_num % 3 == 0:
            self._weather_index = (self._weather_index + 1) % len(self.WEATHER_CYCLE)
            world.weather = self.WEATHER_CYCLE[self._weather_index]

    def apply_emotion_changes(self, emotion: EmotionState, changes: dict) -> None:
        """应用情绪变化"""
        for key, delta in changes.items():
            if hasattr(emotion, key):
                current = getattr(emotion, key)
                setattr(emotion, key, max(0.0, min(100.0, current + delta)))

    def apply_decay(self, emotion: EmotionState) -> None:
        """情绪衰减"""
        for key, rate in self.DECAY_RATES.items():
            current = getattr(emotion, key)
            setattr(emotion, key, max(0.0, current * (1 - rate)))

    def calculate_pad(self, emotion: EmotionState) -> dict:
        """计算 PAD 向量"""
        from services.tts_emotion_engine import compute_pad_from_emotion, classify_mood
        p, a, d = compute_pad_from_emotion(emotion)
        mood = classify_mood(p, a, d)
        return {"pleasure": p, "arousal": a, "dominance": d, "mood": mood}
```

## 五、ConsistencyAgent 详细设计

### 5.1 职责

```
ConsistencyAgent = 质量门

职责：
  → 规则检查（快速，每轮必做）
  → 可选 LLM 检查（慢速，关键轮次做）

规则检查：
  ✅ 情绪突变（|delta| > 30）
  ✅ 对话过短（len < 5）
  ✅ 对话重复（unique_chars / total_chars < 0.3）
  ✅ 事件连续重复

LLM 检查（可选）：
  ✅ 台词是否符合角色性格
  ✅ 台词是否与当前情绪一致
  ✅ 选项是否合理
```

### 5.2 实现

```python
class ConsistencyAgent:
    """一致性 Agent — 输出校验"""

    MAX_EMOTION_DELTA = 30.0

    def __init__(self, llm_tool: LLMTool = None):
        self.llm = llm_tool

    async def check(self, state: AgentState, output: RoundOutput) -> dict:
        """校验输出

        Returns:
            {"passed": bool, "violations": list, "severity": str}
        """
        violations = []

        # 1. 规则检查（必做）
        violations.extend(self._rule_checks(state, output))

        # 2. LLM 检查（可选，关键轮次做）
        if self.llm and state.total_rounds % 5 == 0:  # 每 5 轮做一次
            llm_violations = await self._llm_checks(state, output)
            violations.extend(llm_violations)

        return {
            "passed": len(violations) == 0,
            "violations": violations,
            "severity": "error" if len(violations) >= 3 else ("warning" if violations else "ok"),
        }

    def _rule_checks(self, state: AgentState, output: RoundOutput) -> list:
        """规则检查"""
        violations = []

        # 情绪突变
        for key, delta in state.output_emotion_changes.items():
            if abs(delta) > self.MAX_EMOTION_DELTA:
                violations.append(f"情绪[{key}]变化过大: {delta:.1f}")

        # 对话过短
        if output.character_dialogue and len(output.character_dialogue.strip()) < 5:
            violations.append("对话内容过短")

        # 对话重复
        if output.character_dialogue and self._is_repetitive(output.character_dialogue):
            violations.append("对话内容重复度过高")

        # 事件连续重复
        if len(state.event_history) >= 2:
            last = state.event_history[-1]
            current_type = state.pending_event.get("type") if state.pending_event else None
            if current_type and last.get("type") == current_type:
                violations.append(f"连续重复事件: {current_type}")

        return violations

    async def _llm_checks(self, state: AgentState, output: RoundOutput) -> list:
        """LLM 语义检查"""
        violations = []

        try:
            personality = state.character_personality.get("keywords", [])
            prompt = f"""检查以下对话是否符合角色性格。

角色性格：{'、'.join(personality) if personality else '普通'}
角色台词：{output.character_dialogue}
当前情绪：好感度{state.emotion.favorability:.0f}，快乐{state.emotion.happiness:.0f}

如果有问题，返回具体问题。如果没有问题，返回"无问题"。"""

            result = await self.llm.generate(
                prompt, max_tokens=100, temperature=0.3, call_type="dialogue"
            )

            if result and "无问题" not in result and len(result) > 5:
                violations.append(f"LLM 检查: {result[:50]}")

        except Exception as e:
            logger.warning(f"LLM 检查失败: {e}")

        return violations

    def _is_repetitive(self, text: str) -> bool:
        """检测文本重复度"""
        if len(text) < 10:
            return False
        unique = len(set(text))
        return unique / len(text) < 0.3
```

## 六、Orchestrator 详细设计

### 6.1 职责

```
Orchestrator = 流水线调度员

职责：
  → 记忆读写（Redis/PG）
  → 流水线编排（调用各组件）
  → TTS 语音合成（异步）
  → 图片查询（异步）
  → 事件压缩（事件结束时）
  → 输出构建

不负责：
  → 任何"思考"逻辑（那是 ScriptEngine 的职责）
  → 任何"计算"逻辑（那是 WorldSimulator 的职责）
```

### 6.2 实现

```python
class AgentOrchestrator:
    """Agent 编排器 — 流水线调度员"""

    def __init__(self, text_gen, db_manager, redis_client, scenes_data,
                 image_service, tts_service):
        # 核心组件
        self.script_engine = ScriptEngine(llm_tool=LLMTool(text_gen))
        self.world_simulator = WorldSimulator()
        self.consistency = ConsistencyAgent(llm_tool=LLMTool(text_gen))

        # 记忆存储
        self.redis_store = RedisMemoryStore(redis_client)
        self.pg_store = PgMemoryStore(db_manager)

        # 外部服务
        self.image_service = image_service
        self.tts_service = tts_service

        # 多会话状态
        self._states: Dict[str, AgentState] = {}
        self._lock = threading.Lock()

    async def process_input(self, user_input: str, thread_id: str) -> Dict:
        """主流程"""

        # Step 1: 加载状态
        state = self._states.get(thread_id)
        await self._load_from_redis(state)

        # Step 2: 记录玩家输入
        state.last_player_input = user_input
        state.advance_round()
        state.add_to_history("player", user_input)

        # Step 3: ScriptEngine 生成交互数据
        output = await self.script_engine.generate_round(state)

        # Step 4: WorldSimulator 执行
        self.world_simulator.advance_time(state.world, output.phase)
        self.world_simulator.update_weather(state.total_rounds, state.world)
        self.world_simulator.apply_decay(state.emotion)

        # Step 5: 应用情绪变化（玩家选择后）
        # 注意：这里假设玩家已经在上一轮选择了选项
        # 实际的情绪变化在玩家选择时应用

        # Step 6: ConsistencyAgent 校验
        consistency_result = await self.consistency.check(state, output)

        # Step 7: TTS（异步）
        tts_future = self._generate_tts(state, output.character_dialogue)

        # Step 8: 写回 Redis
        await self._save_to_redis(state, user_input, output.character_dialogue)

        # Step 9: 事件结束处理
        if output.event_ended:
            await self._handle_event_end(state)

        # Step 10: 构建响应
        response = self._build_response(state, output, consistency_result)

        # 等待 TTS
        if tts_future:
            try:
                tts_result = await asyncio.wait_for(tts_future, timeout=5.0)
                if tts_result:
                    response["audio_url"] = tts_result.get("audio_url")
            except (asyncio.TimeoutError, Exception):
                pass

        return response

    def _build_response(self, state: AgentState, output: RoundOutput,
                        consistency: dict) -> dict:
        """构建响应"""
        return {
            "character_dialogue": output.character_dialogue,
            "player_options": [o.__dict__ for o in output.player_options],
            "scene": output.scene,
            "current_states": state.emotion.to_dict(),
            "state_changes": state.output_emotion_changes,
            "phase": output.phase,
            "elapsed_minutes": round(state.world.elapsed_minutes, 1),
            "weather": state.world.weather,
            "current_time": state.world.current_time,
            "round": state.total_rounds,
            "is_game_finished": output.is_game_finished,
            "ending_type": output.ending_type,
            "consistency": consistency,
        }
```
